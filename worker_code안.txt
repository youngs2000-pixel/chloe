export default {
  async fetch(request, env) {
    const corsHeaders = {
      "Access-Control-Allow-Origin": "*",
      "Access-Control-Allow-Methods": "POST, OPTIONS",
      "Access-Control-Allow-Headers": "Content-Type",
      "Content-Type": "application/json",
    };

    if (request.method === "OPTIONS") {
      return new Response(null, { status: 204, headers: corsHeaders });
    }

    if (request.method !== "POST") {
      return new Response(
        JSON.stringify({ text: null, error: "Only POST requests are allowed." }),
        { status: 405, headers: corsHeaders }
      );
    }

    try {
      const payload = await request.json();

      const prompt = `
너는 초등학생이 플레이하는 따뜻한 아기고양이 돌보기 게임의 문장 작가다.
반드시 한국어로 작성한다.
초등학교 5학년도 이해할 수 있게 쉽게 쓴다.
귀엽고 따뜻하고 약간 재미있게 쓴다.
너무 무섭거나 슬프거나 잔인한 표현은 쓰지 않는다.
게임 수치나 별점은 절대 바꾸지 않는다.
오직 화면에 표시할 문장만 작성한다.
2~3문장 이내로 짧게 작성한다.

게임 상황:
${JSON.stringify(payload, null, 2)}

위 상황에 어울리는 아기고양이 반응 또는 평가 문장을 작성해줘.
`;

      const response = await fetch(
        `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${env.GEMINI_API_KEY}`,
        {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
          },
          body: JSON.stringify({
            contents: [
              {
                parts: [{ text: prompt }],
              },
            ],
            generationConfig: {
              temperature: 0.9,
              maxOutputTokens: 180,
            },
          }),
        }
      );

      if (!response.ok) {
        return new Response(
          JSON.stringify({ text: null, error: "Gemini API request failed." }),
          { status: 500, headers: corsHeaders }
        );
      }

      const data = await response.json();
      const text =
        data?.candidates?.[0]?.content?.parts?.[0]?.text?.trim() || null;

      return new Response(JSON.stringify({ text }), {
        status: 200,
        headers: corsHeaders,
      });
    } catch (error) {
      return new Response(
        JSON.stringify({ text: null, error: "AI generation failed." }),
        { status: 500, headers: corsHeaders }
      );
    }
  },
};
